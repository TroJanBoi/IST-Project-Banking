conf t
hostname ML-SW-1

enable secret cisco
line console 0
password pass
login
exit

line vty 0
password pass
transport input telnet
login
exit

vlan 10
name it
vlan 20
name seal

ip routing

interface vlan10
ip add 172.16.10.2 255.255.255.0
st 10 ip 172.16.10.254
st 10 pri 100
st 10 pree
no shutdown

int vlan20
ip add 172.16.20.2 255.255.255.0
st 20 ip 172.16.20.254
st 20 pri 100
st 20 pree
no shut
ex

vtp mode server
vtp domain MyBankNetwork  
vtp version 2
vtp password pass 
 
ip dhcp excluded-address 172.16.10.1 172.16.10.11
ip dhcp pool VLAN10_POOL
network 172.16.10.0 255.255.255.0
default-router 172.16.10.254
dns-server 8.8.8.8

ip dhcp excluded-address 172.16.20.1 172.16.20.11
ip dhcp pool VLAN20_POOL
 network 172.16.20.0 255.255.255.0
 default-router 172.16.20.254
 dns-server 8.8.8.8

interface range Gi1/0/1-24	
switchport mode trunk
int gi1/0/24
no sh
no sw
ip add 172.16.1.2 255.255.255.0


router rip
v 2
network 172.16.1.0
no auto-summary

service dhcp
end
