///multi sw 1
en
conf t
hostname ML-SW-1

enable secret cisco
line console 0
password pass
login
exit

line vty 0
password pass
transport input telnet
login
exit

vlan 10
name ACCOUT
vlan 20
name IT

ip routing

interface vlan10
ip add 192.168.110.2 255.255.255.240
st 10 ip 192.168.110.14
st 10 pri 100
st 10 pree
no shutdown

int vlan20
ip add 192.168.120.2 255.255.255.240
st 20 ip 192.168.120.14
st 20 pri 100
st 20 pree
no shut
ex

vtp mode server
vtp domain MyBankNetwork  
vtp version 2
vtp password pass 
 
ip dhcp excluded-address 192.168.110.14
ip dhcp excluded-address 192.168.110.1 192.168.110.3
ip dhcp pool vlan10
network 192.168.110.0 255.255.255.240
default-router 192.168.110.1
dns-server 8.8.8.8


ip dhcp excluded-address 192.168.120.14 
ip dhcp excluded-address 192.168.120.1 192.168.120.3
ip dhcp pool vlan20
network 192.168.120.0 255.255.255.240
default-router 192.168.120.1
dns-server 8.8.8.8


interface range Gi1/0/1-24	
switchport mode trunk
sw t a v 10,20
int gi1/0/24
no sh
no sw
ip add 192.168.13.2 255.255.255.248


router rip
v 2
network 192.168.13.0
no auto-summary

service dhcp
end
write memory
